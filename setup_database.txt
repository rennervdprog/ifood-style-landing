-- 1. ADICIONAR COLUNAS DE ENTREGA NA TABELA STORES
-- Caso as colunas já existam, o Supabase apenas ignorará o comando ou você pode remover as que já tem.
ALTER TABLE public.stores 
ADD COLUMN IF NOT EXISTS delivery_mode TEXT DEFAULT 'platform',
ADD COLUMN IF NOT EXISTS own_delivery_fee NUMERIC DEFAULT 0,
ADD COLUMN IF NOT EXISTS delivery_fee_type TEXT DEFAULT 'fixed',
ADD COLUMN IF NOT EXISTS delivery_base_km NUMERIC DEFAULT 0,
ADD COLUMN IF NOT EXISTS delivery_fee_base NUMERIC DEFAULT 0,
ADD COLUMN IF NOT EXISTS delivery_fee_per_km NUMERIC DEFAULT 0;

-- 2. GARANTIR POLÍTICAS DE RLS (SEGURANÇA)
-- Habilita o RLS se ainda não estiver habilitado
ALTER TABLE public.stores ENABLE ROW LEVEL SECURITY;

-- Remove políticas antigas para evitar duplicidade ao rodar o script
DROP POLICY IF EXISTS "Lojas são visíveis para todos" ON public.stores;
DROP POLICY IF EXISTS "Lojistas podem atualizar suas próprias lojas" ON public.stores;

-- Cria as políticas de acesso
CREATE POLICY "Lojas são visíveis para todos" 
ON public.stores FOR SELECT 
USING (true);

CREATE POLICY "Lojistas podem atualizar suas próprias lojas" 
ON public.stores FOR UPDATE 
USING (auth.uid() = owner_id) 
WITH CHECK (auth.uid() = owner_id);

-- 3. FORÇAR ATUALIZAÇÃO DO CACHE DO SUPABASE
-- Este comando é essencial para resolver o erro "Could not find column in schema cache"
NOTIFY pgrst, 'reload schema';
